  function parseELF(buf){
    var dv=new DataView(buf),bytes=new Uint8Array(buf),r={ok:false};
    if(buf.byteLength<64||bytes[0]!==0x7F||bytes[1]!==0x45||bytes[2]!==0x4C||bytes[3]!==0x46) return r;
    var bits=bytes[4]===1?32:bytes[4]===2?64:0; if(!bits) return r;
    var le=bytes[5]===1;
    function u16(o){return le?dv.getUint16(o,true):dv.getUint16(o,false);}
    function u32(o){return le?dv.getUint32(o,true):dv.getUint32(o,false);}
    function u64(o){var lo=le?dv.getUint32(o,true):dv.getUint32(o+4,false),hi=le?dv.getUint32(o+4,true):dv.getUint32(o,false);return hi*0x100000000+lo;}
    var osabi=bytes[7],etype=u16(16),machine=u16(18);
    var entry,phoff,shoff,flags,phnum,shnum,shstrndx,shentsize;
    if(bits===32){entry=u32(24);phoff=u32(28);shoff=u32(32);flags=u32(36);phnum=u16(44);shentsize=u16(46);shnum=u16(48);shstrndx=u16(50);}
    else{entry=u64(24);phoff=u64(32);shoff=u64(40);flags=u32(48);phnum=u16(56);shentsize=u16(58);shnum=u16(60);shstrndx=u16(62);}
    // String table for section names
    var strtabOff=-1;
    var shdrSz=bits===32?40:64;
    if(shstrndx<shnum&&shoff>0&&shoff+(shstrndx+1)*shdrSz<=buf.byteLength){
      var ssOff=shoff+shstrndx*shdrSz;
      strtabOff=bits===32?u32(ssOff+16):u64(ssOff+24);
    }
    function readStr(idx){
      if(strtabOff<0) return '';
      var p=strtabOff+idx,e2=p;
      while(e2<bytes.length&&bytes[e2]!==0)e2++;
      try{return new TextDecoder().decode(bytes.slice(p,e2));}catch(e){return '';}
    }
    var sections=[];
    if(shoff>0&&shnum>0){
      for(var i=0;i<Math.min(shnum,64);i++){
        var off=shoff+i*shdrSz;
        if(off+shdrSz>buf.byteLength) break;
        var stype=u32(off+4),sflags,saddr,ssize,snameOff=u32(off);
        if(bits===32){sflags=u32(off+8);saddr=u32(off+12);ssize=u32(off+20);}
        else{sflags=u64(off+8);saddr=u64(off+24);ssize=u64(off+32);}
        if(stype===0&&ssize===0) continue; // skip NULL empty
        sections.push({name:readStr(snameOff)||('section['+i+']'),type:stype,flags:sflags,addr:saddr,size:ssize});
      }
    }
    // Embedded strings
    var strs=[],run='',seen2={};
    for(var k=0;k<Math.min(bytes.length,65536);k++){
      var b=bytes[k];
      if(b>=0x20&&b<0x7F){run+=String.fromCharCode(b);}
      else{if(run.length>=7&&/[a-zA-Z]{3}/.test(run)&&!seen2[run]){seen2[run]=1;strs.push(run);}run='';}
    }
    var kw=['version','gcc','clang','llvm','gnu','linux','lib','main','init','.c','.cpp','Copyright','build'];
    strs=strs.filter(function(s){var l=s.toLowerCase();return kw.some(function(k){return l.indexOf(k)!==-1;})||/\d+\.\d+\.\d+/.test(s);}).slice(0,30);
    return{ok:true,bits:bits,le:le,osabi:osabi,osabiName:ELF_OSABIS[osabi]||'ABI '+osabi,etype:etype,etypeName:ELF_ETYPES[etype]||'Type 0x'+etype.toString(16),machine:machine,machineName:ELF_MACHINES[machine]||'Unknown (0x'+machine.toString(16)+')',entry:entry,flags:flags,phnum:phnum,shnum:shnum,sections:sections,strs:strs};
  }

  // ── Intel HEX parser ──────────────────────────────────────────────
  function parseIntelHex(text){
    var lines=text.split('\n'),records=0,dataBytes=0,extBase=0,entry=null,errors=0;
    var minAddr=0xFFFFFFFF,maxAddr=0,segs=[];
    var curStart=null,curEnd=null;
    for(var i=0;i<lines.length;i++){
      var l=lines[i].trim();
      if(!l) continue;
      if(l[0]!==':'){errors++;continue;}
      var bc=parseInt(l.slice(1,3),16);
      var addr=parseInt(l.slice(3,7),16);
      var rt=parseInt(l.slice(7,9),16);
      var data=[];
      for(var j=0;j<bc;j++) data.push(parseInt(l.slice(9+j*2,11+j*2),16));
      // verify checksum
      var sum=bc+(addr>>8)+(addr&0xFF)+rt;
      for(var j=0;j<data.length;j++) sum+=data[j];
      if(((~sum+1)&0xFF)!==parseInt(l.slice(9+bc*2,11+bc*2),16)) errors++;
      if(rt===0){
        var abs=extBase+addr;
        dataBytes+=bc;records++;
        if(abs<minAddr) minAddr=abs;
        if(abs+bc>maxAddr) maxAddr=abs+bc;
        if(curStart===null){curStart=abs;curEnd=abs+bc;}
        else if(abs<=curEnd+64){curEnd=Math.max(curEnd,abs+bc);}
        else{segs.push({start:curStart,end:curEnd});curStart=abs;curEnd=abs+bc;}
      } else if(rt===1){
        break;
      } else if(rt===2){
        extBase=((data[0]<<8)|data[1])<<4;
      } else if(rt===4){

  // ── Hex viewer ────────────────────────────────────────────────
  function byteHexSpan(b,sep){
    var h=('0'+b.toString(16)).slice(-2);
    var cls=b===0?'hx-00':b===0xFF?'hx-ff':b<0x20||b===0x7F?'hx-lo':b<0x80?'hx-pr':b<0xC0?'hx-mid':'hx-hi';
    return '<span class="'+cls+'">'+h+'</span>'+sep;
  }
  function byteAscSpan(b){
    var cls=b>=0x20&&b<0x7F?'hx-pr':'hx-mid';
    var c=b>=0x20&&b<0x7F?esc(String.fromCharCode(b)):'.';
    return '<span class="'+cls+'">'+c+'</span>';
  }
  function renderHex(buf,name,partial,totalSize,dlHref){
    var bytes=new Uint8Array(buf),rows=[],cols=16,maxBytes=Math.min(bytes.length,4096);
    for(var i=0;i<maxBytes;i+=cols){
      var addr=('00000000'+i.toString(16)).slice(-8);
      var hex='',asc='';
      for(var j=0;j<cols;j++){
        if(i+j<maxBytes){
          var b=bytes[i+j];
          hex+=byteHexSpan(b,j===7?' &ensp; ':' ');
          asc+=byteAscSpan(b);
        } else {
          hex+='   '+(j===7?' &ensp; ':'');
          asc+=' ';
        }
      }
      rows.push('<span class="hx-addr">'+addr+'</span><span class="hx-hex">'+hex+'</span><span class="hx-asc">'+asc+'</span>');
    }
    var shown=maxBytes,total=totalSize||bytes.length;
    var infoTxt=partial?'First '+fmtBytes(shown)+' of '+fmtBytes(total):''+fmtBytes(total);
    var dlLink=dlHref?'<a class="hx-dl" href="'+esc(dlHref)+'" download>↓ Download</a>':'';
    var bar='<div class="hx-bar"><span class="hx-bar-info">'+esc(name||'')+'&ensp;·&ensp;'+infoTxt+'</span>'+dlLink+'</div>';
    return '<div class="hx-wrap">'+bar+'<pre class="hx-pre">'+rows.join('\n')+'</pre></div>';
  }

  function openRaw(href, name){
    loading('Loading…');
    // Fetch only first 64 KB — enough for hex display and text detection
    fetch(href,{headers:{'Range':'bytes=0-65535'}}).then(function(r){
      var partial=r.status===206;
      var totalSize=null;
      var cr=r.headers.get('Content-Range');
      if(cr){var m=cr.match(/\/(\d+)/);if(m)totalSize=parseInt(m[1],10);}
      return r.arrayBuffer().then(function(buf){return{buf:buf,partial:partial,totalSize:totalSize};});
    }).then(function(d){
      var buf=d.buf,bytes=new Uint8Array(buf);
      var sample=bytes.slice(0,Math.min(512,bytes.length));
      var isBinary=false;
      for(var i=0;i<sample.length;i++){
        var b=sample[i];
        if(b===0||(b<9&&b!==0)||(b>13&&b<32)){isBinary=true;break;}
      }
      if(isBinary){
        setContent(renderHex(buf,name,d.partial,d.totalSize,href));
      } else {
        var text=new TextDecoder().decode(buf);
        var note=d.partial&&d.totalSize?'<div style="padding:.3rem 1rem;font-size:.68rem;color:var(--muted);border-bottom:1px solid var(--border)">First 64 KB of '+fmtBytes(d.totalSize)+'</div>':'';
        setContent(note+'<pre class="code-view">'+esc(text)+'</pre>');
      }
    }).catch(function(err){setContent('<div style="padding:1.5rem;color:var(--muted);text-align:center;font-size:.8rem">'+esc(err.message)+'</div>');});
  }

    document.getElementById('btn-list').classList.remove('active');
    filterEntries(searchEl?searchEl.value:'');
  });

  // ── Select mode & ZIP download ────────────────────────────────
  var selectedFiles=new Set();
  var selectMode=false;
  var zipBar=document.getElementById('zip-bar');
  var zipCount=document.getElementById('zip-count');
  var zipDl=document.getElementById('zip-dl');
  var btnSelect=document.getElementById('btn-select');

  // Theme toggle
  var btnTheme=document.getElementById('btn-theme');
  if(btnTheme){
    if(document.body.classList.contains('light')) btnTheme.classList.add('active');
    btnTheme.addEventListener('click',function(){
      var isLight=document.body.classList.toggle('light');
      btnTheme.classList.toggle('active',isLight);
      try{localStorage.setItem('theme',isLight?'light':'dark');}catch(e){}
      if(window._updateOrbs) window._updateOrbs(!isLight);
    });
  }

  function updateZipBar(){
    var n=selectedFiles.size;
    if(n>0){zipBar.classList.add('show');if(zipCount)zipCount.textContent=n+' selected';}
    else{zipBar.classList.remove('show');}
    if(zipDl) zipDl.disabled=(n===0);
  }

  function clearSelection(){
    selectedFiles.clear();
    document.querySelectorAll('.bl-cb').forEach(function(cb){cb.checked=false;});
    document.querySelectorAll('.gc').forEach(function(g){g.classList.remove('gc-selected');});
    updateZipBar();
    selectMode=false;
    if(btnSelect){btnSelect.classList.remove('active');btnSelect.textContent='Select';}
    document.querySelector('.bl-card').classList.remove('has-select');
    var cbAll=document.getElementById('cb-all');if(cbAll)cbAll.checked=false;
  }

  if(btnSelect){
    btnSelect.addEventListener('click',function(){
      selectMode=!selectMode;
      this.classList.toggle('active',selectMode);
      this.textContent=selectMode?'Done':'Select';
      document.querySelector('.bl-card').classList.toggle('has-select',selectMode);
      if(!selectMode) clearSelection();
    });
  }

  document.getElementById('bl-body').addEventListener('change',function(ev){
    if(!ev.target.classList.contains('bl-cb')) return;
    var name=ev.target.dataset.name;
    if(ev.target.checked) selectedFiles.add(name);
    else selectedFiles.delete(name);
    updateZipBar();
  });
  // Prevent checkbox clicks from opening the file preview
  document.getElementById('bl-body').addEventListener('click',function(ev){
    if(ev.target.classList.contains('bl-cb')||ev.target.closest('.bl-cb-col')){
      ev.stopPropagation();
    }
  },true);

  var cbAll=document.getElementById('cb-all');
  if(cbAll){
    cbAll.addEventListener('change',function(){
      var checked=this.checked;
      document.querySelectorAll('#bl-body .bl-cb').forEach(function(cb){
        if(cb.closest('tr').style.display==='none') return;
        cb.checked=checked;
        var name=cb.dataset.name;
        if(name){if(checked)selectedFiles.add(name);else selectedFiles.delete(name);}
      });
      updateZipBar();
    });
  }

  if(document.getElementById('zip-cancel')){
    document.getElementById('zip-cancel').addEventListener('click',clearSelection);
  }
  if(document.getElementById('zip-selall')){
    document.getElementById('zip-selall').addEventListener('click',function(){
      // Select all visible entries (files + dirs) in both list and grid
      entries.forEach(function(e){
        if(e.isParent||e.isAsd) return;
        var tr=rowMap[e.name];
        if(tr&&tr.style.display==='none') return; // filtered out
        selectedFiles.add(e.name);
        var cb=tr?tr.querySelector('.bl-cb'):null;
        if(cb) cb.checked=true;
        var gc=gcMap[e.name];
        if(gc){gc.classList.add('gc-selected');var gcb=gc.querySelector('.gc-cb');if(gcb)gcb.checked=true;}
      });
      updateZipBar();
    });
  }

  if(zipDl){
    zipDl.addEventListener('click',function(){
      if(!selectedFiles.size||!window.JSZip) return;
      var btn=this;
      btn.disabled=true;btn.textContent='Zipping…';
      var zip=new JSZip();
      // Recursively list files in a directory by fetching its autoindex page
      function listDir(href,prefix){
        return fetch(href).then(function(r){return r.text();}).then(function(html){
          var re=/<a href="([^"]+)">([^<]*)<\/a>\s+\S+\s+\S+\s+\S+/g,m2,files=[];
          while((m2=re.exec(html))!==null){
            var h=m2[1],n=decodeURIComponent(h.replace(/\/$/,'').split('/').pop());
            if(h==='../') continue;
            if(h.endsWith('/')) files.push({href:new URL(h,href).href,name:prefix+n+'/',isDir:true});
            else files.push({href:new URL(h,href).href,name:prefix+n,isDir:false});
          }
          var subs=[];
          files.forEach(function(f){
            if(f.isDir) subs.push(listDir(f.href,f.name));
            else subs.push(fetch(f.href).then(function(r){return r.arrayBuffer();}).then(function(buf){zip.file(f.name,buf);}));
          });
          return Promise.all(subs);
        });
      }
      // Auto-include matching .asd files for selected audio files
      var filesToZip=new Set(selectedFiles);
      selectedFiles.forEach(function(name){var a=asdMap[name];if(a)filesToZip.add(a.name);});
      var promises=Array.from(filesToZip).map(function(name){
        var e=entryMap[name];if(!e) return Promise.resolve();
        if(e.isDir) return listDir(e.href,e.name+'/');
        return fetch(e.href).then(function(r){return r.arrayBuffer();}).then(function(buf){zip.file(name,buf);});
      });
      Promise.all(promises).then(function(){
        return zip.generateAsync({type:'blob',compression:'DEFLATE',compressionOptions:{level:6}});
      }).then(function(blob){
        var a=document.createElement('a');
        a.href=URL.createObjectURL(blob);
        var folder=path.replace(/\//g,'_').replace(/^_/,'').replace(/_$/,'');
        a.download=(folder||'files')+'.zip';
        a.click();
        setTimeout(function(){URL.revokeObjectURL(a.href);},10000);
        btn.disabled=false;btn.textContent='↓ ZIP';
      }).catch(function(err){
        btn.disabled=false;btn.textContent='↓ ZIP';
        alert('ZIP failed: '+err.message);
      });
    });
  }

  // Update grid cards after ID3 loads
  var _origApply=applyId3ToRow;
  applyId3ToRow=function(name,tags,dur){
    _origApply(name,tags,dur);
    if(gcMap[name]) updateGridCards();
  };

  // ── Keyboard navigation ─────────────────────────────────────
  var fileEntries=entries.filter(function(e){return !e.isDir&&!e.isParent;});
  document.addEventListener('keydown',function(ev){
    if(ev.target.tagName==='INPUT'||ev.target.tagName==='TEXTAREA'||ev.target.isContentEditable) return;
    if(ev.key==='Escape'){
      if(document.getElementById('bl-main').classList.contains('has-preview')) closePreview();
      return;
    }
    if(ev.key==='ArrowDown'||ev.key==='ArrowUp'){
      ev.preventDefault();
      var curIdx=-1;
      if(activeRow){for(var i=0;i<fileEntries.length;i++){if(rowMap[fileEntries[i].name]===activeRow){curIdx=i;break;}}}
      var nextIdx=ev.key==='ArrowDown'?curIdx+1:curIdx-1;
      if(nextIdx>=0&&nextIdx<fileEntries.length){
        var nxt=fileEntries[nextIdx];
        openPreview(nxt,rowMap[nxt.name]||null);
        var nr=rowMap[nxt.name];if(nr)nr.scrollIntoView({block:'nearest',behavior:'smooth'});
      }
      return;
    }
    // ←/→ navigate through images
    if(ev.key==='ArrowLeft'||ev.key==='ArrowRight'){
      var prevB=document.getElementById('img-prev'),nextB=document.getElementById('img-next');
      if(prevB||nextB){
        ev.preventDefault();
        if(ev.key==='ArrowLeft'&&prevB&&!prevB.disabled) prevB.click();
        if(ev.key==='ArrowRight'&&nextB&&!nextB.disabled) nextB.click();
      }
      return;
    }
    if(ev.key===' '){
      ev.preventDefault();
      var aud=document.querySelector('#preview-content audio');
      if(aud){aud.paused?aud.play().catch(function(){}):aud.pause();}
    }
  });

  // ── Hash routing ──────────────────────────────────────────────
  function openFromHash(){
    var hash=window.location.hash;
    if(!hash||hash==='#') return false;
    var name=decodeURIComponent(hash.slice(1));
    var entry=entryMap[name];
    if(entry&&!entry.isDir&&!entry.isParent){
      openPreview(entry,rowMap[name]||null,true);
      return true;
    }
    return false;
  }

  // Always fetch README in background to set page title
  if(readmeEntry){
    fetch(readmeEntry.href).then(function(r){return r.ok?r.text():null;}).then(function(text){
      if(!text) return;
      var h1m=text.match(/^#+\s+(.+)/m);
      if(h1m){
        var titleH=document.getElementById('bl-title');
        if(titleH) titleH.textContent=h1m[1].trim();
        document.title=h1m[1].trim();
      }
    }).catch(function(){});
  }

  // On load: hash takes priority over README auto-open
  if(!openFromHash()){
    if(readmeEntry){
      setTimeout(function(){openPreview(readmeEntry,rowMap[readmeEntry.name]||null);},200);
    }
  } else {
    // Scroll active row into view
    setTimeout(function(){if(activeRow)activeRow.scrollIntoView({block:'center',behavior:'smooth'});},200);
  }

  // Handle browser back/forward
  window.addEventListener('hashchange',function(){
    if(!window.location.hash||window.location.hash==='#') closePreview();
    else openFromHash();
  });

})();
